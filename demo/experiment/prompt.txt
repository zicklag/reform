# Prompt Parser for Interactive Fiction
#
# Converts player prompts (typed in the REPL) into action facts.
# The REPL in prompt mode turns input into (prompt, word1, word2, ...) facts.
# These rules consume prompt facts and produce action facts for the IF engine.
#
# Built-in shorthands only. All verbs are defined via "understand" in game files.

# Direction shorthands: n, s, e, w
(rule, parse_n,
    ( -(prompt, n) ),
    ( (going, north) )
)
(rule, parse_s,
    ( -(prompt, s) ),
    ( (going, south) )
)
(rule, parse_e,
    ( -(prompt, e) ),
    ( (going, east) )
)
(rule, parse_w,
    ( -(prompt, w) ),
    ( (going, west) )
)

# Look shorthand
(rule, parse_l,
    ( -(prompt, l) ),
    ( (looking) )
)

# Inventory shorthand
(rule, parse_i,
    ( -(prompt, i) ),
    ( (inventorying) )
)

# Examine shorthand
(rule, parse_x,
    ( -(prompt, x, ?item) ),
    ( (examining, ?item) )
)
(rule, parse_x_the,
    ( -(prompt, x, the, ?item) ),
    ( (examining, ?item) )
)

# Help command
(rule, parse_help,
    ( -(prompt, help) ),
    ( (print, Commands: go <dir>, take <item>, drop <item>, look, inventory, where am I, examine <item>, n/s/e/w/l/i/x) )
)
