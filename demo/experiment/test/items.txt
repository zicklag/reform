# Test: Taking and dropping items
$ load ../game.txt

# Should start in the Garden with the key
$ assert (here, Garden)
$ assert (in, key, Garden)

# Take the key
> take key
$ assert (carrying, key)
$ assert not (in, key, Garden)

# Inventory should show the key
> inventory
$ assert (carrying, key)

# Drop the key
> drop key
$ assert (in, key, Garden)
$ assert not (carrying, key)

# Take the key again
> take key
$ assert (carrying, key)

# Go north to Kitchen
> go north
$ assert (here, Kitchen)

# Drop key in Kitchen
> drop key
$ assert (in, key, Kitchen)
$ assert not (carrying, key)

# Try to take something that isn't here
> take book
$ assert not (carrying, book)
