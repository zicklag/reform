# Example 1: Bootstrap Preamble
# 
# This demonstrates how English text can be parsed into facts
# using rules. The bootstrap defines a minimal grammar for
# type definitions, relation definitions, and action definitions.
#
# The key insight: the grammar is just rules. The engine doesn't
# know what "A Person is a being" means — the rules define it.

# --- Step 1: Tokenization ---
# Split text into words. This is the bridge from raw text to facts.

rule:tokenize:text(?s):tokens(?s, ?s):0

# --- Step 2: Pattern matching over tokens ---
# "A X is a Y" → type(X), subtype(X, Y)

rule:parse_is_a:tokens(?s, ?tokens):definition(?s, is_a(?x, ?y))
# Note: this is a simplified version. Real tokenization would split words.

# --- Step 3: Extract definitions from patterns ---

rule:define_type:definition(?s, is_a(?x, ?y)):type(?x), subtype(?x, ?y):0

# --- Step 4: Relation definitions ---
# "A X has a Y" → relation(Y, X)

rule:parse_has_a:tokens(?s, ?tokens):definition(?s, has_a(?x, ?y))

rule:define_relation:definition(?s, has_a(?x, ?y)):relation(?y, ?x):0

# --- Step 5: Action definitions ---
# "A X can V a Y" → action(V, X, Y)

rule:parse_can:tokens(?s, ?tokens):definition(?s, can(?x, ?v, ?y))

rule:define_action:definition(?s, can(?x, ?v, ?y)):action(?v, ?x, ?y):0

# --- Now test the bootstrap ---
# We'll use direct facts for now since tokenization is simplified.
# The patterns above show how the grammar would work.

# Let's define types directly:
type(Person)
type(Thing)
type(Room)
type(String)
type(Number)

# Subtypes:
subtype(Person, being)
subtype(Thing, object)

# Relations:
relation(name, Person)
relation(age, Person)
relation(location, Person)
relation(at, Thing)

# Actions:
action(take, Person, Thing)
action(drop, Person, Thing)
action(go, Person, Room)

# Now define rules that use these types:

# A Person can take a Thing if they are in the same room
rule:take:at(?thing,?room),location(?person,?room):carrying(?person,?thing):0

# A Person can drop a Thing if they are carrying it
rule:drop:carrying(?person,?thing):at(?thing,?room),location(?person,?room):0

# A Person can go to a Room if there is a connection
rule:go:location(?person,?from),connects(?from,?to):location(?person,?to):0

run

# Show the world model
facts
rules
